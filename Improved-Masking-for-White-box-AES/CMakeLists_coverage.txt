CMAKE_MINIMUM_REQUIRED(VERSION 2.8)

PROJECT(WBAES_COVERAGE)

include_directories(include)

# 设置覆盖率编译标志
SET(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -std=c99 -O0 -g --coverage -fprofile-arcs -ftest-coverage")
SET(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} --coverage")

aux_source_directory(src/WBMatrix WBAES_SRC)
aux_source_directory(src/ WBAES_SRC)

add_library(WBAES_LIB ${WBAES_SRC})

# 原始测试可执行文件
add_executable(WBAES test/main.c)
target_link_libraries(WBAES WBAES_LIB)

# 综合测试可执行文件
add_executable(WBAES_COMPREHENSIVE_TEST test/comprehensive_test.c)
target_link_libraries(WBAES_COMPREHENSIVE_TEST WBAES_LIB)

# 单元测试可执行文件
add_executable(WBAES_UNIT_TEST test/unit_test.c)
target_link_libraries(WBAES_UNIT_TEST WBAES_LIB)

# 边界测试可执行文件
add_executable(WBAES_BOUNDARY_TEST test/boundary_test.c)
target_link_libraries(WBAES_BOUNDARY_TEST WBAES_LIB)

# 稳定性测试可执行文件
add_executable(WBAES_STABILITY_TEST test/stability_test.c)
target_link_libraries(WBAES_STABILITY_TEST WBAES_LIB)

# WBMatrix模块测试可执行文件
add_executable(WBAES_WBMATRIX_TEST test/wbmatrix_test.c)
target_link_libraries(WBAES_WBMATRIX_TEST WBAES_LIB)

# 主测试套件可执行文件
add_executable(WBAES_MASTER_TEST test/master_test.c)
target_link_libraries(WBAES_MASTER_TEST WBAES_LIB)

# 覆盖率完善测试可执行文件
add_executable(WBAES_COVERAGE_COMPLETION_TEST test/coverage_completion_test.c)
target_link_libraries(WBAES_COVERAGE_COMPLETION_TEST WBAES_LIB)

# 快速覆盖率测试可执行文件
add_executable(WBAES_QUICK_COVERAGE_TEST test/quick_coverage_test.c)
target_link_libraries(WBAES_QUICK_COVERAGE_TEST WBAES_LIB)

# 最终覆盖率测试可执行文件
add_executable(WBAES_FINAL_COVERAGE_TEST test/final_coverage_test.c)
target_link_libraries(WBAES_FINAL_COVERAGE_TEST WBAES_LIB)
